192  // Get a character, while interpreting entities.
193  // The length can be from 0 to 4 bytes.
194  inline static const TCHAR *GetChar( const TCHAR *p, TCHAR *_value, int* length)
195  {
196  assert( p );
197 
198  *length = 1;
199 
200  if ( *length == 1 )
201  {
202  if ( *p == T('&') )
203  return GetEntity( p, _value, length);
204  *_value = *p;
205  return p+1;
206  }
207  else if ( *length )
208  {
209  //strncpy( _value, p, *length ); // lots of compilers don't like this function (unsafe),
210  // and the null terminator isn't needed
211  for( int i=0; p[i] && i<*length; ++i ) {
212  _value[i] = p[i];
213  }
214  return p + (*length);
215  }
216  else
217  {
218  // Not valid text.
219  return 0;
220  }
221  }
